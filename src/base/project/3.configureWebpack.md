# 三.比例尺

:::demo

```vue
<template>
  <div class="map-example">
    <div ref="map" class="map"></div>
  </div>
</template>

<script>
export default {
  name: "scale",
  mounted() {
    let {
      Map,
      View,
      layer: { Tile: TileLayer },
      source: { OSM },
      control: { defaults, ScaleLine },
    } = ol
    new Map({
      view: new View({
        center: [12579156, 3274244], // 坐标
        zoom: 12, // 放大倍数
      }),
      layers: [
        new TileLayer({
          // 创建一个使用Open Street Map地图源的瓦片图层
          source: new OSM(),
        }),
      ],
      controls: defaults(/* 地图默认控件 */).extend([new ScaleLine()]),
      target: this.$refs.map,
    })
  },
}
</script>

<style lang="sass" scoped>
.map-example {
  width: 100%;
  height: 80vh;
  position relative
  .map {
    width: 100%;
    height: 100%;
    border: 1px solid $accentColor;
  }
}
</style>
```
