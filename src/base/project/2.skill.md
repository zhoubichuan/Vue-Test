# 二.鼠标位置

:::demo

```vue
<template>
  <div class="map-example">
    <div ref="map" class="map"></div>
    <div class="position" ref="position"></div>
  </div>
</template>

<script>
export default {
  name: "mousePosition",
  mounted() {
    let {
      Map,
      View,
      layer: { Tile: TileLayer },
      source: { OSM },
      control: { defaults, MousePosition },
      coordinate: { createStringXY },
    } = ol
    new Map({
      view: new View({
        center: [12579156, 3274244], // 坐标
        zoom: 12, // 放大倍数
      }),
      layers: [
        new TileLayer({
          // 创建一个使用Open Street Map地图源的瓦片图层
          source: new OSM(),
        }),
      ],
      controls: defaults({
        /* 地图默认控件 */
        attributionOptions: {
          collapsible: true, // 右下角详细信息默认收缩
        },
      }).extend([
        new MousePosition({
          coordinateFormat: createStringXY(3), // 保留 3 位小数
          projection: "EPSG:4326", // 输出 4326 投影的坐标，默认3857
          target: this.$refs.position,
        }),
      ]),
      target: this.$refs.map,
    })
  },
}
</script>

<style lang="sass" scoped>
.map-example {
  width: 100%;
  height: 80vh;
  position relative
  .map {
    width: 100%;
    height: 100%;
    border: 1px solid $accentColor;
  }
  .position{
    position absolute
    top -50px
    right 0
    white-space: nowrap
  }
}
</style>
```
